ls - listar
cd - muda de diretório
pwd - mostra o caminho do diretório que estou
ls -a exibe arquivos ocultos
mkdir - cria pasta nova
cat - diz o que tem dentro do arquivo
nano (arquivo.extensao) - cria arquivo e escreve nele
rm - remove arquivo
rm -rf - remove de forma recursiva e forçada todos os diretórios
